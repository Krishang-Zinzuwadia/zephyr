                    if self.ui_overlay:
                        self.ui_overlay.update_transcription(result.text, is_final=True)
                    
                    # Show completion animation
                    if self.ui_overlay:
                        self.ui_overlay.show_completion()
                
                # Finalize input
                if self.input_simulator:
                    self.input_simulator.finalize_input()
